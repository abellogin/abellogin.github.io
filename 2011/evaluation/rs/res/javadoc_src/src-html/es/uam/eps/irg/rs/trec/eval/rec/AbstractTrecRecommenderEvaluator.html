157        /**

158         * {@inheritDoc}

159         */

160        public void evaluate() {

161            final List<PrintStream> recStreams = new ArrayList<PrintStream>();

162            for (int i = 0; i < recommenders.size(); i++) {

163                String rec = recommenderNames.get(i);

164                if ((!new File(rec).exists()) || (new File(rec).exists() && overwriteOutputFile)) {

165                    try {

166                        recStreams.add(new PrintStream(new File(rec)));

167                    } catch (Exception e) {

168                        e.printStackTrace();

169                        recStreams.add(null);

170                    }

171                } else {

172                    System.err.println("Ignoring " + rec);

173                    recStreams.add(null);

174                }

175            }

176            // check if qrels file exists

177            if (qrelFile != null) {

178                if (new File(qrelFile).exists()) {

179                    System.err.println("Qrels file " + qrelFile + " will be rewritten");

180                    // and reset it

181                    try {

182                        new PrintStream(new File(qrelFile)).close();

183                    } catch (Exception e) {

184                        e.printStackTrace();

185                    }

186                }

187            }

188            // check if relevant file exists

189            PrintStream psRel = null;

190            if (relFile == null) {

191                relFile = getRelevantItemsFileName();

192            }

193            if (new File(relFile).exists()) {

194                System.err.println("Relevant file " + relFile + " will be rewritten");

195            }

196            try {

197                psRel = new PrintStream(new File(relFile));

198            } catch (Exception e) {

199                e.printStackTrace();

200            }

201    

202            ExecutorService userPool = null;

203            if (userThreads > 1) {

204                userPool = Executors.newFixedThreadPool(userThreads);

205            }

206            for (long userID : test.keySet()) {

207                // for each user

208                // get a subset of her test set, i.e., those rated high

209                // (depending on method 'isValidItemForTestSet')

210                final Map<Long, Float> relMap = getRelevantMap(userID);

211                if (relMap == null) {

212                    continue;

213                }

214                final Set<Long> relSet = getRelevantSet(userID);

215                if ((relSet == null) || (relSet.isEmpty())) {

216                    continue;

217                }

218                // select N unrated items by this user

219                final Map<Long, Set<Long>> notRelMap = getNotRelevantMap(userID, relSet);

220                if ((notRelMap == null) || (notRelMap.isEmpty())) {

221                    continue;

222                }

223                // write qrels

224                if (qrelFile != null) {

225                    try {

226                        writeQrelFile(userID, relMap, qrelFile);

227                    } catch (Exception e) {

228                        e.printStackTrace();

229                    }

230                }

231                // write relevant file

232                if (psRel != null) {

233                    try {

234                        writeRelevantItems(userID, relMap, psRel);

235                    } catch (Exception e) {

236                        e.printStackTrace();

237                    }

238                }

239                final UserEvaluator ue = new UserEvaluator(userID, relSet, notRelMap, recommenders, recommenderNames, recStreams, recThreads);

240                if (userPool == null) {

241                    ue.run();

242                } else {

243                    userPool.execute(ue);

244                }

245            }

246            if (userPool != null) {

247                userPool.shutdown();

248                try {

249                    while (!userPool.awaitTermination(10 * 24 * 60 * 60, TimeUnit.SECONDS)) {

250                        System.err.println("Try again!");

251                    }

252                } catch (InterruptedException e) {

253                    e.printStackTrace();

254                }

255            }

256            // close files

257            for (int i = 0; i < recommenders.size(); i++) {

258                final PrintStream writer = recStreams.get(i);

259                if (writer == null) {

260                    continue;

261                }

262                writer.close();

263            }

264            if (psRel != null) {

265                psRel.close();

266            }

267        }

269        private synchronized void estimatePreference(long userID, long itemID, final Recommender recommender, final Map<Long, Float> itemPreferenceMap) {

270            float pref = Float.NaN;

271            try {

272                pref = recommender.estimatePreference(userID, itemID);

273            } catch (NoSuchElementException nsee) {

274            } catch (NoSuchUserException nsue) {

275            } catch (TasteException te) {

276            }

277            if (Double.isNaN(pref)) {

278                if (wrappingRec != null) {

279                    try {

280                        pref = wrappingRec.estimatePreference(userID, itemID);

281                    } catch (NoSuchElementException nsee) {

282                    } catch (NoSuchUserException nsue) {

283                    } catch (TasteException te) {

284                    }

285                }

286            }

287            // NaN is then sorted higher than any other value, and thus it should be removed

288            if (!Double.isNaN(pref)) {

289                itemPreferenceMap.put(itemID, pref);

290            }

291        }

293        /**

294         *

295         * Returns the set of relevant items for a particular user.

296         * By default it uses the output of #getRelevantMap(long),

297         * by returning the keySet of that map.

298         *

299         * @param userID user id

300         * @return set of relevant items for a user

301         */

302        protected Set<Long> getRelevantSet(long userID) {

303            return getRelevantMap(userID).keySet();

304        }

305    

306        /**

307         *

308         * Returns the map of relevant items for a particular user.

309         * The map contains items in the keys, and their corresponding relevance

310         * values in the values.

311         *

312         * @param userID user id

313         * @return the map of relevant items for a user, if userID is not in the test set, an empty map is returned

314         */

315        protected Map<Long, Float> getRelevantMap(long userID) {

316            if (test.containsKey(userID)) {

317                return test.get(userID);

318            } else {

319                return new HashMap<Long, Float>();

320            }

321        }

323        /**

324         * Returns the set of not relevant items for a particular user.

325         * This method depends on each evaluation methodology.

326         *

327         * @param userID user id

328         * @return set of not relevant items for a user

329         */

330        protected abstract Set<Long> getNotRelevantSet(long userID);

331    

332        /**

333         * Returns the set of not relevant items for a particular user and a particular item.

334         * This method depends on each evaluation methodology.

335         * By default, it returns the same set of items for every relevant item for a user.

336         *

337         * @param userID user id

338         * @param items set of relevant items

339         * @return set of not relevant items for a user

340         */

341        protected Map<Long, Set<Long>> getNotRelevantMap(long userID, Set<Long> items) {

342            Map<Long, Set<Long>> map = new HashMap<Long, Set<Long>>();

343            Set<Long> notRelSet = getNotRelevantSet(userID);

344            for (long i : items) {

345                map.put(i, notRelSet);

346            }

347            return map;

348        }

350        /**

351         *

352         * A generic implementation for generating a qrel file.

353         * For each relevant item (#getRelevantMap(long)) a line in the file

354         * is included with the following pattern:

355         * userID \t 0 \t itemID \t relevanceValue

356         *

357         * @param userID user id

358         * @param relMap map of relevant items for a user

359         * @param qrelFile name of the file

360         * @throws java.io.FileNotFoundException

361         */

362        protected void writeQrelFile(long userID, Map<Long, Float> relMap, String qrelFile) throws FileNotFoundException {

363            PrintStream writerQrels = new PrintStream(new FileOutputStream(qrelFile, true));

364            for (Map.Entry<Long, Float> e : relMap.entrySet()) {

365                long itemID = e.getKey();

366                int rel = e.getValue().intValue();

367                writerQrels.println(userID + "\t0\t" + itemID + "\t" + rel);

368            }

369            writerQrels.close();

370        }

373         *

374         * For each relevant item (#getRelevantMap(long)) a line in the file

375         * is included with the following pattern:

376         * userID \t itemID \t relevanceValue

377         *

378         * @param userID user id

379         * @param relMap map of relevant items for a user

380         * @param ps stream of the file

381         * @throws java.io.FileNotFoundException

382         */

383        protected void writeRelevantItems(long userID, Map<Long, Float> relMap, PrintStream ps) throws FileNotFoundException {

384            for (Map.Entry<Long, Float> e : relMap.entrySet()) {

385                long itemID = e.getKey();

386                int rel = e.getValue().intValue();

387                ps.println(userID + "\t" + itemID + "\t" + rel);

388            }

389        }

392            String f = null;

393            for (String r : recommenderNames) {

394                File ff = new File(r);

395                f = ff.getParent() + "/";

396                break;

397            }

398            return f;

401        /**

402         *

403         * A generic implementation for generating a ranking given a user and

404         * her list of preferences.

405         * For each preference included in userItemPreferenceMap (previously filled

406         * with values generated by some recommender) a ranking is generated.

407         * This ranking may contains less items than those in the union of relSet

408         * and notRelSet, if additional constraints are included (#setMaxRankingSize(int)).

409         * For each preference, a line is included in the file:

410         * userID \t Q0 \t itemID \t position \t preferenceValue \t r

411         *

412         * @param userID user id

413         * @param relSet set of relevant items for a user

414         * @param notRelMap map of not-relevant items for a user

415         * @param userItemPreferenceMap map containing the preferences inferred by some recommender for a user

416         * @param recName name of the recommender which generated set of relevant items for a user

417         * @param writer stream where the ranking will be written

418         */

419        protected synchronized void writeRanking(long userID, final Set<Long> relSet, final Map<Long, Set<Long>> notRelMap, final Map<Long, Float> userItemPreferenceMap, String recName, final PrintStream writer) {

420            final Map<Float, Set<Long>> preferenceMap = new HashMap<Float, Set<Long>>();

421            for (Entry<Long, Float> e : userItemPreferenceMap.entrySet()) {

422                addItemToRanking(e.getKey(), e.getValue(), preferenceMap);

423            }

424            // Sort items by estimated preference

425            final List<Float> sortedScores = new ArrayList<Float>(preferenceMap.keySet());

426            Collections.sort(sortedScores, Collections.reverseOrder());

427            // Write estimated preferences

428            int pos = 1;

429            for (float pref : sortedScores) {

430                for (long itemID : preferenceMap.get(pref)) {

431                    writer.println(userID + "\tQ0\t" + itemID + "\t" + pos + "\t" + pref + "\t" + "r");

432                    pos++;

433                }

434                if (pos > maxRankSize) {

435                    break;

436                }

437            }

438        }

440        /**

441         *

442         * Convenience method which encapsulates some trivial operations over the current map of preferences

443         *

444         * @param item current item

445         * @param pref preference towards an item (for some -unknown- user)

446         * @param preferenceMap mapping of preferences for some user

447         */

448        protected static void addItemToRanking(long item, float pref, Map<Float, Set<Long>> preferenceMap) {

449            Set<Long> items = preferenceMap.get(pref);

450            if (items == null) {

451                items = new TreeSet<Long>();

452                preferenceMap.put(pref, items);

453            }

454    

455            items.add(item);

456        }

473        private class UserRecommenderEvaluator extends TimerTask {

474    

475            private long userID;

476            private Set<Long> relSet;

477            private Map<Long, Set<Long>> notRelMap;

478            private PrintStream writer;

479            private Recommender r;

480            private String rec;

481            private Map<Long, Float> userItemPreferenceMap;

482    

483            public UserRecommenderEvaluator(long userID, Set<Long> relSet, Map<Long, Set<Long>> notRelMap, PrintStream writer, Recommender r, String rec, Map<Long, Float> userItemPreferenceMap) {

484                this.userID = userID;

485                this.relSet = relSet;

486                this.notRelMap = notRelMap;

487                this.writer = writer;

488                this.r = r;

489                this.rec = rec;

490                this.userItemPreferenceMap = userItemPreferenceMap;

491            }

492    

493            @Override

494            public void run() {

495                for (long item : relSet) {

496                    estimatePreference(userID, item, r, userItemPreferenceMap);

497                    for (long itemN : notRelMap.get(item)) {

498                        estimatePreference(userID, itemN, r, userItemPreferenceMap);

499                    }

500                }

501                writeRanking(userID, relSet, notRelMap, userItemPreferenceMap, rec, writer);

502            }

503        }

505        private class UserEvaluator extends TimerTask {

506    

507            private long userID;

508            private Set<Long> relSet;

509            private Map<Long, Set<Long>> notRelMap;

510            private List<Recommender> recs;

511            private List<String> recNames;

512            private List<PrintStream> streams;

513            private int nThread;

514    

515            public UserEvaluator(long userID, Set<Long> relSet, Map<Long, Set<Long>> notRelMap, List<Recommender> recs, List<String> recNames, List<PrintStream> writer, int nThread) {

516                this.userID = userID;

517                this.relSet = relSet;

518                this.notRelMap = notRelMap;

519                this.recs = recs;

520                this.recNames = recNames;

521                this.streams = writer;

522                this.nThread = nThread;

523            }

524    

525            @Override

526            public void run() {

527                ExecutorService recPool = null;

528                if (nThread > 1) {

529                    recPool = Executors.newFixedThreadPool(nThread);

530                }

531                for (int i = 0; i < recs.size(); i++) {

532                    final PrintStream writer = streams.get(i);

533                    if (writer == null) {

534                        continue;

535                    }

536                    final Recommender r = recs.get(i);

537                    final String rec = recNames.get(i);

538                    final Map<Long, Float> userItemPreferenceMap = new HashMap<Long, Float>();

539                    final UserRecommenderEvaluator ue = new UserRecommenderEvaluator(userID, relSet, notRelMap, writer, r, rec, userItemPreferenceMap);

540                    if (recPool == null) {

541                        ue.run();

542                    } else {

543                        recPool.execute(ue);

544                    }

545                }

546                if (recPool != null) {

547                    recPool.shutdown();

548                    try {

549                        while (!recPool.awaitTermination(10 * 24 * 60 * 60, TimeUnit.SECONDS)) {

550                            System.err.println("Try again!");

551                        }

552                    } catch (InterruptedException e) {

553                        e.printStackTrace();

554                    }

555                }

556            }

557        }
